    document.write(haresh);

* nested loop with array of array :-

(Example - 1) :-

using for loop & print index wise data multiple row in array.

var haresh = [
    ["harry", 18, "male", "B.Com"],
    ["sunny", 19, "male", "BCA"],
    ["sarah", 18, "female", "BCA"],
    ["tom", 17, "male", "B.A."],
];

for(var a = 0; a < 4; a++){
    document.write(a + " ");
    for(var b = 0; b < 4; b++){
        document.write(haresh[a][b] + " ");
    }
    document.write("<br>");
}

(output) :-

0 harry 18 male B.Com

1 sunny 19 male BCA

2 sarah 18 female BCA

3 tom 17 male B.A.

* using nested loop with table wise data of array :-

(Example - 1) :-

using nested loop & print index wise multiple row in array.

var haresh = [
    ["harry", 18, "male", "B.Com"],
    ["sunny", 19, "male", "BCA"],
    ["sarah", 18, "female", "BCA"],
    ["tom", 17, "male", "B.A."],
];

document.write("<table>");
for(var a = 0; a < 4; a++){
    document.write("<tr>");
    for(var b = 0; b < 4; b++){
        document.write("<td>" + haresh[a][b] + "</td>");
    }
    document.write("</tr>");
}
document.write("</table>");

(output) :-

5

* nested loop with automatic count length of array :-

(Example - 1) :-

using for loop & pr int index wise multiple row in array.

var haresh = [
    ["harry", 18, "male", "B.Com"],
    ["sunny", 19, "male", "BCA"],
    ["sarah", 18, "female", "BCA"],
    ["tom", 17, "male", "B.A."],
    ["cat", 20, "male", "BBA"]
];

document.write(haresh.length);

for(var a = 0; a < haresh.length; a++){
    document.write(a + " ");
    for(var b = 0; b < haresh[a].length; b++){
        document.write(haresh[a][b] + " ");
    }
    document.write("<br>");
}

(output) :-

harry 18 male B.Com

sunny 19 male BCA

sarah 18 female BCA

tom 17 male B.A.

cat 20 male BBA
