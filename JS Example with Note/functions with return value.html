* function with return value *

- એક function ની અંદર parameter + parameter property ની value pass કરીને function ની અંદર એક variable declair કરીને parameter ની property ને concatination કરીને નીચે તે variable ને return કરે છે. function ના parameter ની value નીચે declair કરેલા variable માં store થાય છે, અને તે variable ની value function માં return(store) થાય છે, અને તે function નીચે એક variable માં declair કરાવીને parameter ની value update કરીએ તેને print કરીને output show થાય છે.



* function with return value structure :- *

function functionName(parameter1, parameter2) {
   variable = parameter1 + " " + parameter2;
   return variable
}
var fn = functionName(argument1, argument2);
   document.write(fn);


(1) Parameter with set default value :-

(Example - 1) :-

function abc(fname ="value1", lname ="value2") {
   var a = "hello" + fname + "" + lname;
   return a;
}

var fn = abc("value3", "value4");
   document.write(fn);

(output) :-

hello value3 value4

(Example - 2) :-

function abc(fname ="value1", lname ="value2") {
   var a = "hello" + fname + "" + lname;
   return a;
}

var fn = abc(three = "value3", four = "value4");
   document.write(fn);

(output) :-

hello value3 value4

(Example - 3) :-

function abc(fname ="value1", lname ="value2") {
   var a = "hello" + fname + "" + lname;
   return a;
}

var fn = abc();
   document.write(fn);

(output) :-

(Example - 4) :-

(2) Make a Formula Calculate With Return Value :-

-પહેલા function ની અંદર ૩ parameter pass કરીને function ની અંદર એક variable declair કરીને ઉપરના parameter ને concatination કરીને નીચે તે variable ને return કરાવી દીધો. બીજા function માં એક parameter લઇ function ની અંદર એક variable માં ઉપરનો parameter pass કરીને તેને જેટલા mark હતા એની સાથે parameter ને divide(/) કરીને તેને 100 સાથે multiplication(ગુણાકાર) કરી તેને concatination કરી % ની sign use કરી જેથી તે output માં number ની પાછળ % print કરશે, અને તે variable ને તે જ function માં print કરાવી નીચે બીજો એક variable લઇ તેમાં function call કરાવીને તેમાં argument pass કરાવી, પછી એક second function લઇ તેમાં ઉપરના variable ને function માં call કરાવી દીધો.

function sum(math, eng, sci) {
   var i = math + eng + sci;
   return i;
}

function percentage(tt) {
   var per = tt/300 * 100 + " %";
   document.write(per);
}

var total = sum(63,25,50);
   percentage(total);

(output) :-

46 %
